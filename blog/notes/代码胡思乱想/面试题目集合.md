## 防抖：

    ```-js
    //防抖：n秒内只触发一次，如果变化过快则清除计时器
        function debounce(func,wait) {
          let timeout;
          return function () {
            let context = this;
            let args = arguments;

            if (timeout) clearTimeout(timeout);

            let callNow = !timeout;
            timeout = setTimeout(() => {
              timeout = null;
            }, wait)

            if (callNow) func.apply(context, args)
          }
        }
    ```

## 节流：

    ```-js
    //节流：触发过快时没n秒执行一次
            function throttle(func, wait) {
              let timeout;
              return function() {
                let context = this;
                let args = arguments;
                if (!timeout) {
                  timeout = setTimeout(() => {
                    timeout = null;
                    func.apply(context, args)
                  }, wait)
                }
              }
            }
    ```
